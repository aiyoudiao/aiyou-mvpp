layui.use(["okCookie"], function () {
    let $ = layui.jquery;
    

    // 创建cookie
    $.cookie("hello", "word")

    // 读取cookie
    let hello = $.cookie("hello");
    console.log(hello);

    // 删除cookie
    $.cookie("hello", null);
});
